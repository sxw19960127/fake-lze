# 基本使用

`ES5`中的仿类结构，构造器：

```js
function Dog(name,age) {
	this.name = name;
	this.age = age;
}
Dog.prototype.bark = function() {
	console.log(`${this.name}`)
}
const lili = new Dog("lili",2);
const wangcai = new Dog("wangcai",1);
```

# 类

```js
class Dog{
	constructor(name,age) {
		this.name = name;
		this.age = age;
	} // 注意,不用加逗号
	bark() { // 定义原型上的方法直接写就行 
		console.log(`${this.name}`)
	}
}
const lili = new Dog("lili",2);
const wangcai = new Dog("wangcai",1);
```

注意事项：

- 类声明不会被提升
- 类声明中的代码都自动运行在严格模式下
- 调用类必须使用`new`
- 类中所有的方法都是不可枚举的
- 类中的方法不能用`new`调用
- 类的方法中重写类名会报错

```js
// ES5模拟实现类语法
const Dog = (function() {
	"use strct";
	const Dog = function(name,age) {
		if(!(this instanceof Dog)) {
			throw new Error("必须用new去调用方法")
		}
		this.name = name;
		this.age = age;
	}
	Object.defineProperty(Dog.prototype,"bark",{
		value: function() {
			if(this instanceof Dog.prototype.bark) {
				throw new Error("不能使用new调用")
			}
			console.log(`${this.name}`)
		},
		enumerable: false
	})
	return Dog;
})()
const lili = new Dog("lili",2);
const wangcai = new Dog("wangcai",1);
for(var prop in lili) {
	console.log(prop) // 遍历出两个属性
}
```

`getter`与`setter`，设置读取属性：

```js
class Dog{
	constructor(name,age) {
		this.name = name;
		this.age = age;
	} 
	bark() { 
		console.log(`${this.name}`)
	}
	static showInfo() {
		console.log("这是一条狗")
	}
	get info() {
		return `我叫${this.name},我${this.age}了`
	}
	set nickName(value) { // 设置一个值
		this.nick = value;
	}
	get nickName() { // 获取一个值
		return this.nick;
	}
}
```

类的继承：

```js
// ES5类的继承
function() Animal(name) {
	this.name = name;
	this.thirsty = 100;
	this.food = [];
}
Animal.prototype.drink = function() {
	return this.thirsty -= 10;
}
Animal.prototype.eat = function(item) {
	this.food.push(item);
}
function Dog(name, breed) {
	Animal.call(this.name);
	this.breed = breed;
}

// wangcai.constructor = Dog; new Animal().constructor = Animal
// 创建一个空对象,将Dog的原型指向这个空对象
Dog.prototype = Object.create(Animal.prototype); 
// 将Dog的原型指向Dog自身
Dog.prototype.constructor = Dog; 
Dog.prototype.bark = function() {
	console.log("狗在叫")
}
// wangcai身上具有3个方法函数可以被调用
var wangcai = new Dog("wangcai", "中华田园犬"); 
```

```js
// ES6中的继承 
class Animal{
	constructor(name) {
		this.name = name;
		this.thirsty = 100;
		this.food = []
	}
	drink() {
		return this.thirsty -= 10
	}
	eat(item) {
		this.food.push(item)
	}
}
class Dog extends Animal{
	constructor(name, breed) {
        // 调用super()目的是让程序走一遍constructor里面的代码
		super(name); 
		this.breed = breed
	}
	bark() {
		console.log("狗在叫")
	}
}
var wangcai = new Dog("wangcai", "中华田园犬"); 
```

```js
// 静态方法的继承

// 直接可以Dog.Info去调用父级上的静态方法;
```

```js
class Coder{
    // 类里面都是函数方法
    name(val) {
        console.log(val);
        return val; // 切记要 return,否则是 undefined
    }
    skill(val) {
        console.log(this.name('舒小伟') + ':' + 'skill' + val)
    }
    // 给类进行传参,使用 constructor 进行类的参数设置
    constructor(a,b) {
        this.a = a;
        this.b = b;
    }
    add() {
        return this.a + this.b
    }
}
// 步骤1,2
let sxw = new Coder;
// sxw.name('舒小伟');
sxw.skill('web')

// 步骤3
// 给类进行传参
let sxw = new Coder(1,2);
console.log(sxw.add())
```
